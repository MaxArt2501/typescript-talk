Conditional types

 
T extends U ? X : Y
 
type Exclude<T, U> = T extends U ? never : T

type NonNullable<T> = Exclude<T, null | undefined>
 
interface UserWithId {
  id: string;
  name: string;
}
type UserKeys = keyof UserWithId; // 'id' | 'name'

type User = {
  [K in Exclude<UserKeys, 'id'>]: UserWithId[K];
}; // { name: string }
